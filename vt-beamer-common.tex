%
% colorBar - Draw a colored box labeled with the color
% arguments: 1 - color to show (used as background)
%            2 - color for text label (foreground)
\newcommand{\colorBar}[2]{%
  \setbeamercolor{myColor}{fg=#2,bg=#1}
  \begin{beamercolorbox}[wd=\textwidth,ht=2ex,dp=1ex,center]{myColor}
      #1
  \end{beamercolorbox}
}
